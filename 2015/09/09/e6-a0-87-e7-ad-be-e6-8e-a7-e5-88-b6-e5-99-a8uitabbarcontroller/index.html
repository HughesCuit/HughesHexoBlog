标签控制器UITabBarController

- (BOOL)application:(UIApplication *)application didFinishLaunchingWithOptions:(NSDictionary *)launchOptions {
    _window = [[UIWindow alloc] initWithFrame:[UIScreen mainScreen].bounds];
    _window.backgroundColor = [UIColor whiteColor];

    UITabBarController *tbc = [[UITabBarController alloc] init];
    tbc.tabBar.barTintColor = [UIColor colorWithRed:163/255.0f green:63/255.0f blue:58/255.0f alpha:1];
    tbc.tabBar.tintColor = [UIColor whiteColor];
    tbc.tabBar.translucent = NO;

    ViewController *vc = [[ViewController alloc] init];
    vc.title = @"乐行于蜀";
    vc.tabBarItem.image = [UIImage imageNamed:@"icon_menu_hp_press"];
    vc.tabBarItem.selectedImage = [UIImage imageNamed:@"icon_menu_hp_selected"];
    UINavigationController *nc = [[UINavigationController alloc] initWithRootViewController:vc];
    nc.navigationBar.barTintColor = [UIColor colorWithRed:163/255.0f green:63/255.0f blue:58/255.0f alpha:1];
    nc.navigationBar.tintColor = [UIColor whiteColor];
    nc.navigationBar.translucent = NO;
    nc.navigationBar.titleTextAttributes = @{
                                             NSFontAttributeName:[UIFont boldSystemFontOfSize:24],
                                             NSForegroundColorAttributeName:[UIColor whiteColor]
                                             };

    ViewController1 *vc1 = [[ViewController1 alloc] init];
    vc1.title = @"推荐";
    vc1.tabBarItem.image = [UIImage imageNamed:@"icon_menu_x_press"];
    vc1.tabBarItem.selectedImage = [UIImage imageNamed:@"icon_menu_x_selected"];
    UINavigationController *nc1 = [[UINavigationController alloc] initWithRootViewController:vc1];
    nc1.navigationBar.barTintColor = [UIColor colorWithRed:163/255.0f green:63/255.0f blue:58/255.0f alpha:1];
    nc1.navigationBar.tintColor = [UIColor whiteColor];
    nc1.navigationBar.translucent = NO;
    nc1.navigationBar.titleTextAttributes = @{
                                             NSFontAttributeName:[UIFont boldSystemFontOfSize:24],
                                             NSForegroundColorAttributeName:[UIColor whiteColor]
                                             };

    ViewController2 *vc2 = [[ViewController2 alloc] init];
    vc2.title = @"景点";
    vc2.tabBarItem.image = [UIImage imageNamed:@"icon_menu_profile_press"];
    vc2.tabBarItem.selectedImage = [UIImage imageNamed:@"icon_menu_profile_selected"];
    UINavigationController *nc2 = [[UINavigationController alloc] initWithRootViewController:vc2];
    nc2.navigationBar.barTintColor = [UIColor colorWithRed:163/255.0f green:63/255.0f blue:58/255.0f alpha:1];
    nc2.navigationBar.tintColor = [UIColor whiteColor];
    nc2.navigationBar.translucent = NO;
    nc2.navigationBar.titleTextAttributes = @{
                                             NSFontAttributeName:[UIFont boldSystemFontOfSize:24],
                                             NSForegroundColorAttributeName:[UIColor whiteColor]
                                             };

    ViewController3 *vc3 = [[ViewController3 alloc] init];
    vc3.title = @"更多";
    vc3.tabBarItem.image = [UIImage imageNamed:@"icon_menu_ihave_press"];
    vc3.tabBarItem.selectedImage = [UIImage imageNamed:@"icon_menu_ihave_selected"];
    UINavigationController *nc3 = [[UINavigationController alloc] initWithRootViewController:vc3];
    nc3.navigationBar.barTintColor = [UIColor colorWithRed:153/255.0f green:63/255.0f blue:58/255.0f alpha:1];
    nc3.navigationBar.tintColor = [UIColor whiteColor];
    nc3.navigationBar.translucent = NO;
    nc3.navigationBar.titleTextAttributes = @{
                                             NSFontAttributeName:[UIFont boldSystemFontOfSize:24],
                                             NSForegroundColorAttributeName:[UIColor whiteColor]
                                             };

    tbc.viewControllers = @[nc,nc1,nc2,nc3];

    _window.rootViewController = tbc;

    [_window makeKeyAndVisible];
    return YES;
}
