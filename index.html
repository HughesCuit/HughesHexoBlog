UIScrollView滚动视图

//
//  ViewController.m
//  Lesson9_10_滚动视图
//
//  Created by rimi on 15/9/11.
//  Copyright (c) 2015年 HugheX. All rights reserved.
//

#import "ViewController.h"
#define TOTAL 10

@interface ViewController ()<UIScrollViewDelegate>

@property (nonatomic,strong) UIScrollView *scrollView; /**< 滚动视图 */
@property (nonatomic,strong) UIPageControl *pageControl; /**< 分页视图 */

- (void)initializeUserInterface;/**< 初始化用户界面 */

@end

@implementation ViewController

- (void)initializeUserInterface/**< 初始化用户界面 */
{
    self.view.backgroundColor = [UIColor whiteColor];
    self.title = @"滚动视图";
    //设置导航栏是否透明
    //屏幕坐标系 = translucent == YES ? (0,0) : (0,64);
    self.navigationController.navigationBar.translucent = NO;
    // 添加滚动视图
    [self.view addSubview:self.scrollView];
    //添加图片视图
//    UIImageView *imageView = [[UIImageView alloc] initWithFrame:CGRectMake(0, 0, 575, 1000)];
//    imageView.image = [UIImage imageNamed:@"4"];
//    [self.scrollView addSubview:imageView];
    for(int i = 0; i < 10; i++)
    {
        UIImageView *imgV = [[UIImageView alloc] initWithFrame:CGRectMake(i*CGRectGetWidth(self.view.bounds), 0, CGRectGetWidth(self.scrollView.bounds), CGRectGetHeight(self.scrollView.bounds))];
        imgV.image = [UIImage imageNamed:[NSString stringWithFormat:@"%d",i]];
        //设置边框
        [imgV.layer setBorderWidth:1.0];
        [self.scrollView addSubview:imgV];
    }
    // 添加分页控件
    [self.view addSubview:self.pageControl];
}

#pragma mark - Responds methods
- (void)respondsToPageControl:(UIPageControl*)pageControl{
    pageControl.currentPage =(NSInteger) _scrollView.contentOffset.x / CGRectGetWidth(_scrollView.bounds);
}

#pragma mark - UIScrollViewDelegate methods
// 滚动视图正在滚动中
- (void)scrollViewDidScroll:(UIScrollView *)scrollView{
    _pageControl.currentPage =(NSInteger) scrollView.contentOffset.x / CGRectGetWidth(scrollView.bounds);
}
// 将要拖动
- (void)scrollViewWillBeginDragging:(UIScrollView *)scrollView{
    NSLog(@"%@",NSStringFromSelector(_cmd));
}

- (void)scrollViewDidEndDragging:(UIScrollView *)scrollView willDecelerate:(BOOL)decelerate{
    NSLog(@"%@",NSStringFromSelector(_cmd));
}
// 将要减速
- (void)scrollViewWillBeginDecelerating:(UIScrollView *)scrollView{
    NSLog(@"%@",NSStringFromSelector(_cmd));
}
// 减速完成
- (void)scrollViewDidEndDecelerating:(UIScrollView *)scrollView{
    NSLog(@"%@",NSStringFromSelector(_cmd));
}
// 停止滚动
- (void)scrollViewDidEndScrollingAnimation:(UIScrollView *)scrollView{
    NSLog(@"%@",NSStringFromSelector(_cmd));
}

#pragma mark - Getter methods

- (UIScrollView *)scrollView{
    if (!_scrollView) {
        _scrollView = [[UIScrollView alloc] initWithFrame:CGRectMake(0, 0, CGRectGetWidth(self.view.bounds), CGRectGetHeight(self.view.bounds) - 64)];
        _scrollView.backgroundColor = [UIColor redColor];
        //设置显示内容的大小
        //如果文本内容的宽度大于滚动视图的宽度，高度等于滚动视图的高度，是一个横向滚动的视图；反之则是纵向滚动视图。
        _scrollView.contentSize = CGSizeMake(TOTAL * CGRectGetWidth(self.view.bounds), CGRectGetHeight(_scrollView.bounds));
        //设置代理
        _scrollView.delegate = self;
        //设置是否回弹，默认为YES
        _scrollView.bounces = NO;
        //设置是否允许翻页
        _scrollView.pagingEnabled = YES;
        //设置是否点击状态栏返回顶部
        _scrollView.scrollsToTop = YES;
    }
    return _scrollView;
}

- (UIPageControl *)pageControl{
    if (!_pageControl) {
        _pageControl = [[UIPageControl alloc]init];
        _pageControl.bounds = CGRectMake(0, 0, CGRectGetWidth(self.view.bounds), 30);
        _pageControl.center = CGPointMake(CGRectGetMidX(self.view.bounds), CGRectGetHeight(self.view.bounds) - CGRectGetMidY(_pageControl.bounds)-100);
        // 设置总页数
        _pageControl.numberOfPages = TOTAL;
        // 设置当前显示页
        _pageControl.currentPage = 0;
        // 设置总页数的指示色
        _pageControl.pageIndicatorTintColor = [UIColor redColor];
        // 设置当前页数的指示色
        _pageControl.currentPageIndicatorTintColor = [UIColor blueColor];
        // 添加时间监听
        [_pageControl addTarget:self action:@selector(respondsToPageControl:) forControlEvents:UIControlEventValueChanged];

    }
    return _pageControl;
}

- (void)viewDidLoad {
    [super viewDidLoad];

    [self initializeUserInterface];

}

- (void)didReceiveMemoryWarning {
    [super didReceiveMemoryWarning];
    // Dispose of any resources that can be recreated.
}

@end
